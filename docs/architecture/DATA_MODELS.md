# Data Models — All Shared Types

This document describes every type in `src/types/`. These types are the contracts between modules. If you change a type, you must update every module that uses it.

---

## geometry.ts

```typescript
/** 2D point in document space */
export interface Point {
  x: number;
  y: number;
}

/** 2D vector (same shape as Point but semantically different) */
export type Vector = Point;

/** Axis-aligned bounding box */
export interface BBox {
  x: number;      // top-left x
  y: number;      // top-left y
  width: number;
  height: number;
}

/** A single segment of a path */
export type CurveSegment =
  | { type: 'line'; from: Point; to: Point }
  | { type: 'cubic'; from: Point; cp1: Point; cp2: Point; to: Point }
  | { type: 'quadratic'; from: Point; cp: Point; to: Point }
  | { type: 'arc'; center: Point; rx: number; ry: number; startAngle: number; endAngle: number; rotation: number };

/** A closed or open path made of segments */
export interface Path {
  segments: CurveSegment[];
  closed: boolean;
}
```

## scene.ts

```typescript
import { Point, BBox, Path } from './geometry';

/** All possible scene object types */
export type SceneObjectType = 'rectangle' | 'ellipse' | 'star' | 'freehand' | 'svg-import' | 'raster-import';

/** Base fields shared by all scene objects */
export interface SceneObjectBase {
  id: string;                  // unique, generated by utils/uid.ts
  type: SceneObjectType;
  position: Point;             // top-left corner in document space
  width: number;               // bounding box width in document units
  height: number;              // bounding box height in document units
  rotation: number;            // degrees, clockwise. MVP can default to 0.
  visible: boolean;            // if false, excluded from particle computation
  locked: boolean;             // if true, cannot be selected/moved/resized
}

/** Rectangle shape */
export interface RectangleObject extends SceneObjectBase {
  type: 'rectangle';
}

/** Ellipse shape */
export interface EllipseObject extends SceneObjectBase {
  type: 'ellipse';
}

/** Star shape */
export interface StarObject extends SceneObjectBase {
  type: 'star';
  points: number;              // number of star points (5, 6, 8, etc.)
  innerRadiusRatio: number;    // 0-1, ratio of inner radius to outer radius
}

/** Freehand drawn curve */
export interface FreehandObject extends SceneObjectBase {
  type: 'freehand';
  path: Path;                  // smoothed bezier path from perfect-freehand
}

/** Imported SVG (flattened to paths) */
export interface SVGImportObject extends SceneObjectBase {
  type: 'svg-import';
  paths: Path[];               // extracted and flattened paths
  originalSVG?: string;        // raw SVG string for reference (optional)
}

/** Imported raster image */
export interface RasterImportObject extends SceneObjectBase {
  type: 'raster-import';
  imageDataURL: string;        // base64 data URL of the image
  brightnessGrid?: number[][]; // precomputed brightness map (0-1), computed by rasterSampler
}

/** Union type of all scene objects */
export type SceneObject =
  | RectangleObject
  | EllipseObject
  | StarObject
  | FreehandObject
  | SVGImportObject
  | RasterImportObject;
```

## particles.ts

```typescript
import { Point } from './geometry';

/** Direction particles spawn relative to shape outlines */
export type SpawnDirection = 'inside' | 'outside' | 'both';

/** Falloff curve type for edge density */
export type FalloffType = 'linear' | 'exponential' | 'gaussian';

/** Global particle configuration (applies to entire scene) */
export interface ParticleConfig {
  count: number;               // target total particle count (e.g. 5000)
  minSize: number;             // minimum particle radius in document units
  maxSize: number;             // maximum particle radius
  color: string;               // hex color string, e.g. '#ffffff'
  opacityRandomize: boolean;   // if true, each particle gets random opacity 0.2-1.0
  baseOpacity: number;         // if opacityRandomize is false, all particles use this (0-1)
  spawnDirection: SpawnDirection;
  falloffType: FalloffType;
  falloffDistance: number;     // how far from the curve particles spread (document units)
  falloffBias: number;         // 0-1, controls density: 0 = uniform, 1 = all on the curve
  seed: number;                // random seed for reproducible layouts
}

/** A single computed particle (output of the engine, input to renderer) */
export interface Particle {
  x: number;                   // document space
  y: number;                   // document space
  radius: number;              // document units
  opacity: number;             // 0-1
}

/** Default config to use when initializing the store */
export const DEFAULT_PARTICLE_CONFIG: ParticleConfig = {
  count: 3000,
  minSize: 1,
  maxSize: 3,
  color: '#ffffff',
  opacityRandomize: true,
  baseOpacity: 0.8,
  spawnDirection: 'outside',
  falloffType: 'gaussian',
  falloffDistance: 40,
  falloffBias: 0.5,
  seed: 42,
};
```

## tools.ts

```typescript
import { Point } from './geometry';

/** Available tool types */
export type ToolType = 'select' | 'rectangle' | 'ellipse' | 'star' | 'freehand';

/** Transient state during a tool operation (e.g., dragging to draw a rectangle) */
export interface ToolState {
  activeTool: ToolType;
  isDrawing: boolean;          // true while pointer is down during shape creation
  drawStart: Point | null;     // where the current draw started (document space)
  drawCurrent: Point | null;   // current pointer position (document space)
  shiftHeld: boolean;          // true = constrain proportions
  selectedObjectIds: string[]; // IDs of currently selected scene objects
}
```

## Key Design Decisions

1. **All positions are in document space.** Never store screen or canvas coordinates on objects.
2. **Particles are recomputed, not persisted.** The `Particle[]` array is ephemeral — derived from scene objects + particle config. It's not in a store.
3. **SceneObject is a discriminated union on `type`.** Use `switch(obj.type)` for exhaustive handling.
4. **ParticleConfig is global, not per-object.** All shapes share the same particle settings. This is a deliberate MVP simplification.
5. **The `seed` field enables reproducibility.** Same objects + same config + same seed = same particle layout. Important for export consistency.
